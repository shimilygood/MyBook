知识练习
知识点1. >> $provide.provider方法：【依赖注入】 这种方式并不太常用
$provide.provider:是一种自定义服务的方法，$provide还提供了很多简便的方法，这些方法可以直接被module所引用 
var myBookCtrls = angular.module('myBookCtrls', [],function($provide){
	//可以自定义多个服务,可以通过依赖注入到所有的控制器内
	$provide.provider('sevice',function(){
		this.$get=function(){
			return{
				message:"该服务的名称是：sevice"
			}
		}
	});
	$provide.provider('sevice2',function(){
		this.$get=function(){
			return{
				message:"该服务的名称是：sevice2"
			}
		}
	});
	
});
	
//依赖注入到控制器内
myBookCtrls.controller('studyCtrl3',['$scope','sevice',
	function($scope,sevice){
		console.log(sevice.message);
	}
]);
知识点2. >> $provide.factory方法：【依赖注入】 
var myBookCtrls = angular.module('myBookCtrls', [],function($provide){
	//自定义服务,可以依赖注入到所有的控制器内
	$provide.factory('factory1',function(){
		//可以返回任何数据类型
		return [1,3,4,5,23,45];
	});
});
//依赖注入控制器内
myBookCtrls.controller('studyCtrl3',['$scope','factory1',
	function($scope,factory1){
		console.log(factory1);
	}
]);
对应的快捷声明方法：myBookCtrls.factory()
知识点3. >> $provide.service：【依赖注入】 
var myBookCtrls = angular.module('myBookCtrls', [],function($provide){
	//自定义服务 ---返回的必须是对象
	$provide.service('service1',function(){
		return {title:"sdssd"};
	});
});
//依赖注入控制器内
myBookCtrls.controller('studyCtrl3',['$scope','service1',
	function($scope,service1){
		console.log(service1.title);
	}
]);
对应的快捷声明方法：myBookCtrls.service()
factory 和 service 的区别：
factory可以返回任何数据类型。
service只可以返回对象数据类型。
知识点3. >> 多个控制器内共享数据
factory和service，通过依赖注入，可以实现数据的共享。
（$$prevSiblling、$$nextSiblling）还有一种方法，是通过查找上一个或者下一个sope作用域来实现相邻控制器间的数据共享。
